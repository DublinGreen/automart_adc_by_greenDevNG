const cloudinary = require('cloudinary');
const cloudinaryStorage = require('multer-storage-cloudinary');

  cloud_name: 'xxx', 
  api_key: 'xxx',

});

var parser = multer({ 
  storage: cloudinaryStorage({
    cloudinary: cloudinary,
    folder: 'xxx',
    filename: function (req, file, cb) {
      cb(undefined, file.originalname);
    }
  })
});
